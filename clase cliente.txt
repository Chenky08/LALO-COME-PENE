#pragma once
#include "string"
class Cliente
{
public:
	Cliente(){}
	Cliente(int dni) {
		nombre = "";
		this->dni = dni;
	}
	Cliente(string nombre,int dni){
		this->nombre = nombre;
		this->dni = dni;
	}
	~Cliente(){}


	string getNombre() {
		return nombre;
	}
	void setNombre(string nombre) {
		this->nombre = nombre;
	}

	int getDni() {
		return dni;
	}
	void setDni(int dni) {
		this->dni = dni;
	}

private:
	string nombre;
	int dni;

};
