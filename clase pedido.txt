#pragma once
#include "Cliente.h"

class Pedido
{
public:
	Pedido(){}
	Pedido(int num,string producto,float precio,int cantidad,Cliente* cliente) {
		this->num = num;
		this->producto = producto;
		this->precio = precio;
		this->cantidad = cantidad;
		this->cliente = cliente;
	}
	~Pedido(){}

	int getNum() {
		return num;
	}
	void setNum(int num) {
		this->num = num;
	}

	string getProducto() {
		return producto;
	}
	void setProducto(string producto) {
		this->producto = producto;
	}

	float getPrecio() {
		return precio;
	}
	void setPrecio(float precio) {
		this->precio = precio;
	}

	int getCantidad() {
		return cantidad;
	}
	void setCantidad(int cantidad) {
		this->cantidad = cantidad;
	}

	Cliente* getCliente() {
		return cliente;
	}
	void setCliente(Cliente* cliente) {
		this->cliente = cliente;
	}


private:
	int num;
	string producto;
	float precio;
	int cantidad;
	Cliente* cliente;
};
